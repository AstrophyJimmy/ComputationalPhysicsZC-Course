Hamiltonian for 1D Problems¶

$$H\psi_n = E\psi_n = \dfrac{2\psi_n - \psi_{n+1} - \psi_{n-1} + 2{\Delta x}^2 V_n \psi_n }{2{\Delta x}^2}$$$$\hat{H_n} =\dfrac{1}{{\Delta x}^2} \big(1+\Delta x^2 V_n \big) $$$$\hat{H_{n+1}} =-\dfrac{1}{2{\Delta x}^2}$$$$\hat{H_{n-1}} =-\dfrac{1}{2{\Delta x}^2}$$

dx = 0.1
x=np.round(np.arange(0.7,5,dx),3)
lx = x.size

V = np.zeros(lx)
epsilon = 10
sigma = 1

phi0 = np.zeros(lx)
const=3


for i in range(lx):
    V[i] = 4*epsilon*(np.power(sigma/x[i],12) - np.power(sigma/x[i], 6))
    if x[i]>=1 and x[i]<4:
        phi0[i] = 3

def Hamiltonian(V):
    H = np.zeros([lx,lx])
    for i in range(lx):    
        H[i,i] = 1/np.power(dx,2)+ V[i]
        if lx-i-1: H[i,i+1] = -1/(2*np.power(dx,2))
        if i: H[i,i-1] = -1/(2*np.power(dx,2))
    return H
