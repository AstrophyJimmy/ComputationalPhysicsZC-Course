N = 2500
xs = np.arange(-40,41,1)
ys = np.copy(xs)
lx = xs.size

lattice = np.zeros([lx,lx])
x0 = np.where(xs==0)[0][0]

lattice[x0,x0]=1
for n in range(N):
    x,y = np.nonzero(lattice)   
    inds=[]
    for i,j in zip(x,y):
        if i!=0:
            if not lattice[i-1,j]: inds.append((i-1,j))
        if i!=lx-1:
            if not lattice[i+1,j]: inds.append((i+1,j))
        if j!=0:
            if not lattice[i,j-1]: inds.append((i,j-1))
        if j!=lx-1:
            if not lattice[i,j+1]: inds.append((i,j+1))
    indsUnique = list(set(inds))
    lattice[choice(indsUnique)]=1
    if not n%50:
        #print(indsUnique)
        print("At iteration #"+str(n)+", we have "+str(len(indsUnique))+" sites.")
plt.matshow(lattice)
plt.title("Eden Cluster for N="+str(N))
